package usk.eng.komputer.gotravel;

import android.app.Activity;
import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;

public class Main2Register extends Activity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {

        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main2_register);

        Button tombol3 = (Button) findViewById(R.id.button4);
        tombol3.setOnClickListener(new View.OnClickListener() {

            @Override
            public void onClick(View v) {
                Intent i = new Intent(Main2Register.this, Main2Login.class);
                startActivity(i);

            }
        });
    }
}
